CREATE TABLE PESSOA (ID  SERIAL NOT NULL, NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE USUARIO (CODIGOUSUARIO  SERIAL NOT NULL, CPF VARCHAR(255), EMAIL VARCHAR(255), FONE VARCHAR(255), NOMEUSUARIO VARCHAR(255), SENHA VARCHAR(255), ENDERECO_CODENDERECO INTEGER, PRIMARY KEY (CODIGOUSUARIO))
CREATE TABLE PRODUTO (IDPRODUTO  SERIAL NOT NULL, NOMEPRODUTO VARCHAR(255), QUANTIDADEPRODUTO INTEGER, VALORPRODUTO FLOAT, PEDIDO_CODIGOPEDIDO INTEGER, PRIMARY KEY (IDPRODUTO))
CREATE TABLE PEDIDO (CODIGOPEDIDO  SERIAL NOT NULL, VALORPEDIDO FLOAT, ENDERECOENTREGA_CODENDERECO INTEGER, USUARIO_CODIGOUSUARIO INTEGER, PRIMARY KEY (CODIGOPEDIDO))
CREATE TABLE TELEFONE (ID  SERIAL NOT NULL, NUMERO VARCHAR(255), PESSOA_ID INTEGER, PRIMARY KEY (ID))
CREATE TABLE ENDERECO (CODENDERECO  SERIAL NOT NULL, BAIRRO VARCHAR(255), CIDADE VARCHAR(255), NUMERO VARCHAR(255), RUA VARCHAR(255), PRIMARY KEY (CODENDERECO))
ALTER TABLE USUARIO ADD CONSTRAINT FK_USUARIO_ENDERECO_CODENDERECO FOREIGN KEY (ENDERECO_CODENDERECO) REFERENCES ENDERECO (CODENDERECO)
ALTER TABLE PRODUTO ADD CONSTRAINT FK_PRODUTO_PEDIDO_CODIGOPEDIDO FOREIGN KEY (PEDIDO_CODIGOPEDIDO) REFERENCES PEDIDO (CODIGOPEDIDO)
ALTER TABLE PEDIDO ADD CONSTRAINT FK_PEDIDO_USUARIO_CODIGOUSUARIO FOREIGN KEY (USUARIO_CODIGOUSUARIO) REFERENCES USUARIO (CODIGOUSUARIO)
ALTER TABLE PEDIDO ADD CONSTRAINT FK_PEDIDO_ENDERECOENTREGA_CODENDERECO FOREIGN KEY (ENDERECOENTREGA_CODENDERECO) REFERENCES ENDERECO (CODENDERECO)
ALTER TABLE TELEFONE ADD CONSTRAINT FK_TELEFONE_PESSOA_ID FOREIGN KEY (PESSOA_ID) REFERENCES PESSOA (ID)
